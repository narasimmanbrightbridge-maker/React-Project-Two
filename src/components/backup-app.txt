import { useEffect, useMemo, useState } from "react";
import { Swiper, SwiperSlide } from "swiper/react";
import { Pagination, Navigation, Autoplay, A11y } from "swiper/modules";
import "swiper/css";
import "swiper/css/navigation";
import "swiper/css/pagination";

import { Link } from "react-router-dom";

type Product = {
  id?: number | string;
  name: string;
  image?: string;
  priceOne?: string | number;
  priceTwo?: string | number;
  price?: string | number;
};

type RelatedProductProps = {
  currentProductName?: string;
};

const slugifyProductName = (name: string) =>
  name.toLowerCase().trim().replace(/\s+/g, "-");

function RelatedProduct({ currentProductName }: RelatedProductProps) {
  const [products, setProducts] = useState<Product[]>([]);
  const [isLoading, setIsLoading] = useState(true);
  const [hasError, setHasError] = useState(false);

  useEffect(() => {
    let isMounted = true;
    setIsLoading(true);
    setHasError(false);

    fetch("http://localhost:8000/post")
      .then((res) => {
        if (!res.ok) throw new Error("Failed To Load");
        return res.json();
      })
      .then((data) => {
        if (!isMounted) return;

        const incomingProducts = Array.isArray(data?.post)
          ? data.post
          : Array.isArray(data)
            ? data
            : [];

        setProducts(incomingProducts);
      })
      .catch((err) => {
        console.error("Failed To Fetch Data:", err);
        if (isMounted) setHasError(true);
      })
      .finally(() => {
        if (isMounted) setIsLoading(false);
      });

    return () => {
      isMounted = false;
    };
  }, []);

  const filteredProducts = useMemo(() => {
    if (!currentProductName) return products;

    return products.filter(
      (item) => slugifyProductName(item.name) !== slugifyProductName(currentProductName)
    );
  }, [currentProductName, products]);

  if (isLoading) {
    return (
      <section className="m-10">
        <h1 className="mb-10 text-3xl font-bold">Related Products</h1>
        <p>Loading related products...</p>
      </section>
    );
  }

  if (hasError) {
    return (
      <section className="m-10">
        <h1 className="mb-10 text-3xl font-bold">Related Products</h1>
        <p>Unable to load related products right now.</p>
      </section>
    );
  }

  if (filteredProducts.length === 0) return null;

  return (
    <section className="m-10">
      <h1 className="font-bold text-3xl mb-10">Related Products</h1>
      <div className=" mx-auto">
        <Swiper
          modules={[Pagination, Navigation, Autoplay, A11y]}
          spaceBetween={20}
          slidesPerView={1}
          pagination={{ clickable: true }}
          autoplay={{ delay: 3000 }}
          breakpoints={{
            768: { slidesPerView: 2 },
            1024: { slidesPerView: 4 }
          }}
        >
          {filteredProducts.map((item, index) => (
            <SwiperSlide key={item.id || index}>
              <div className="cards_sec p-5 bg-white rounded-3xl shadow-sm borde">
                <div className="img_sec">
                  <img
                    className="w-full h-48 object-contain m-auto"
                    src={item.image}
                    alt={item.name}
                  />
                </div>
                <div className="cont_sec mt-5">
                  <h2 className="font-bold text-left pb-3 text-xl line-clamp-1">
                    {item.name}
                  </h2>
                  <div className="pricetags flex gap-3 mb-5 items-center">
                    <span className="font-bold text-gray-400 line-through text-sm">
                      ₹{item.priceOne ?? item.price}
                    </span>
                    <span className="font-bold text-2xl text-[#000]-600">
                      ₹{item.priceTwo ?? item.priceOne}
                    </span>
                  </div>
                  <Link
                    to={`/productsdetails/${slugifyProductName(item.name)}`}
                    state={{ product: item }}
                    className="block w-full rounded-2xl bg-[#5a684e] py-3 text-center text-white transition-colors"
                  >
                    View Product
                  </Link>
                </div>
              </div>
            </SwiperSlide>
          ))}
        </Swiper>
      </div>
    </section>
  );
}

export default RelatedProduct;
